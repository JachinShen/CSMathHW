
\begin{exercise}
   Suppose the network $(G,s,t,c)$ has a $k$-layering. Show that ${\rm dist}(s,t) \geq k$.
   That is, every $s$-$t$-path in $G$ has at most $k$ edges.
\end{exercise}

\begin{exercise}
   Conversely, suppose ${\rm dist}(s,t) = k$. Show that $(G,s,t,c)$ has a $k$-layering.
\end{exercise}
